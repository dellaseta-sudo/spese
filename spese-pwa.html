Codice Apps Script 
function doGet(e) {
  return handleRequest(e);
}
function doPost(e) {
  return handleRequest(e);
}
function handleRequest(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getActiveSheet();
  var action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : '');
  var result;

  if (action === 'read') {
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var rows = data.slice(1).map(function(row) {
      return {
        timestamp: row[0] ? row[0].toString() : '',
        data: row[1] ? Utilities.formatDate(new Date(row[1]), Session.getScriptTimeZone(), 'yyyy-MM-dd') : '',
        descrizione: row[2] || '',
        categoria: row[3] || '',
        importo: parseFloat(row[4]) || 0,
        mezzo: row[5] || '',
        persona: row[6] || ''
      };
    });
    result = { status: 'ok', rows: rows };

  } else if (action === 'write') {
    var body = JSON.parse(e.postData.contents);
    var now = new Date();
    sheet.appendRow([
      Utilities.formatDate(now, Session.getScriptTimeZone(), 'dd/MM/yyyy HH.mm.ss'),
      body.data,
      body.descrizione,
      body.categoria,
      body.importo,
      body.mezzo,
      body.persona
    ]);
    result = { status: 'ok' };

  } else if (action === 'delete') {
    var body2 = JSON.parse(e.postData.contents);
    var allData = sheet.getDataRange().getValues();
    for (var i = 1; i < allData.length; i++) {
      var rowTs = allData[i][0] ? allData[i][0].toString() : '';
      if (rowTs === body2.timestamp) {
        sheet.deleteRow(i + 1);
        break;
      }
    }
    result = { status: 'ok' };
  } else {
    result = { status: 'error', message: 'Unknown action' };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
